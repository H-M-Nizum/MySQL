# This function contain (WHEN condition THEEN result)
# And finally ELSE like if-else

SELECT x, y, z, CASE
  WHEN x+y>z AND x+z>y AND y+z>x 
  THEN 'Yes'
  ELSE 'No'
  END AS triangle
FROM Triangle;
