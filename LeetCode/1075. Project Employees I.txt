WITH cte AS
(
  SELECT p.project_id, p.employee_id, experience_years
  FROM Employee AS e
  INNER JOIN Project AS p
  ON e.employee_id = p.employee_id
)

SELECT project_id, ROUND(AVG(experience_years), 2) AS average_years
FROM cte
GROUP BY project_id;
